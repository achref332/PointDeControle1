Début

  // Initialisation des compteurs
  longueur <- 0
  mots <- 1
  voyelles <- 0

  Afficher "Entrez une phrase qui se termine par un point :"
  Lire caractère c

  Tant que c != '.'
      longueur <- longueur + 1

      Si c = ' ' alors
          mots <- mots + 1
      Fin Si

      Si c est une voyelle (a, e, i, o, u, A, E, I, O, U) alors
          voyelles <- voyelles + 1
      Fin Si

      Lire caractère c
  Fin Tant que

  // Ajouter le point final à la longueur
  longueur <- longueur + 1

  // Affichage des résultats
  Afficher "Longueur de la phrase : ", longueur
  Afficher "Nombre de mots : ", mots
  Afficher "Nombre de voyelles : ", voyelles

Fin
